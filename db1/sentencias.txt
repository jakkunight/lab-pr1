# 1er ejercicio

SELECT PERS_NOMBRE || ' ' || PERS_APELLIDO AS NOMBRE_COMPLETO, PERS_SERIE_CI, PERS_CI FROM personas

# 2do ejercicio

SELECT PERS_NOMBRE || ' ' || PERS_APELLIDO AS NOMBRE_COMPLETO, PERS_SERIE_CI, PERS_CI FROM personas WHERE PERS_ID_PAIS = 1 AND PERS_ID_CIUDAD = 10

# 3er ejercicio

SELECT PERS_NOMBRE || ' ' || PERS_APELLIDO AS NOMBRE_COMPLETO, PERS_SERIE_CI, PERS_CI FROM personas WHERE PERS_NRO_FUNCIONARIO IS NOT NULL

# 4to ejercicio

SELECT PERS_NOMBRE || ' ' || PERS_APELLIDO AS NOMBRE_COMPLETO, PERS_SERIE_CI, PERS_CI FROM personas WHERE PERS_FECHA_NACIMIENTO >= DATE '2000-01-01' ORDER BY PERS_FECHA_NACIMIENTO ASC
